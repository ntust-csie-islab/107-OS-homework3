🙂 使用情境說明(包含流程圖)

這次的作業中，我實做了一個叫做PC的應用程式，這個程式有三個主要功能

  1. 讓使用者能夠在priority scheduling跟round robin scheduling之間做切換
  2. 讓使用者可以查看目前所有process的state
  3. 在使用priority scheduling時，可以改變proccess的priority

流程圖:

😇 Demo畫面

首先，先產生三個process，再來輸入PC進入 Process Controller，進入後有四個選項：

  1. Change Scheduling Process
  2. Show all process state
  3. Change process priority
  4. exit

我們先輸入’2’來看目前所以 process的state

一開始default的設定是round robin scheduling，所以process會輪流使用CPU

接著輸入’1’來更改scheduling policy，然後把pid＝4的process的priority從5改成10，pid＝6的process的priority從5改成8

pid=4的process就會從running變成runnable，pid=6還是runnable，pid=8變成running

最後，可以把scheduling policy從priority scheduling切回round robin

輸入’4’可以離開應用程，輸入PC可以再次啟動應用程式

🏃 實作過程(修改哪些檔案[含圖片])

因為xv6沒有scanf()這個function，所以必須自己實做出一個來用

以下是scanf.c的程式碼：










































































#include "types.h"
#include "stat.h"
#include "user.h"

static void
getc(int fd, char ** buffer){
	int  i = 0;
	char buf[256];	// 讀進buffer
	while(read(fd, &buf[i], 1)){
		if((buf[i] == '\n' || buf[i] == ' ' || buf[i] == '\t' || buf[i] == '\r' || buf[i] == '\v' || buf[i] == '\f')){
			break;
		}
		else{
			i++;
		}
	}
	buf[i] = '\0';
	*(buffer) = buf;
}

int atoi_decimal(char *buf){	
	int res = 0;  
	int sign = 1;  
	int i = 0;  
  
	if (buf[0] == '-'){	// 如果是負數，就更新sign
		sign = -1;  
		i++;  
	}
      	for (; buf[i] != '\0'; i++){
    		if(buf[i] - '0' < 0 || buf[i] - '0' > 9)  // 如果不是數字就break
    			break ;
        res = res*10 + buf[i] - '0';
    	}
	return sign*res;
}

void scanf(int fd, char * fmt, ...){
	int i = 0;
	char c;
	char * buf = "";	// 暫時保留input
	int count_args = 1;
	uint var = *((uint*)(void*)&fmt + count_args);
	count_args++;
	for(i = 0; fmt[i]; i++){
		c = fmt[i] & 0xff;
		if(c=='c'){
			getc(fd, &buf);
			*(char *)(var) = buf[0];	
			buf = "";
			var = *((uint*)(void*)&fmt + count_args);
			count_args++;
		}
				
		if(c=='d'){
			getc(fd,&buf);
			*(int *)(var) = atoi_decimal(buf);	
			buf = "";
			var = *((uint*)(void*)&fmt + count_args);
			count_args++;
		}

		if(c=='s'){
			getc(fd, &buf);
			strcpy((char *)var, buf); 		
			var = *((uint*)(void*)&fmt + count_args);
			count_args++;
			buf = "";	
		}
			
	}
	
}

在proc.c中修改scheduler()























































void
scheduler(void)
{
  struct proc *p;
  struct proc *p1;
  
  struct cpu *c = mycpu();
  c->proc = 0;
  
  for(;;){
    // Enable interrupts on this processor.
    sti();

    //highp records the process with the highest priority
    struct proc *highP = 0;
   
    
    // Loop over process table looking for process to run.
    acquire(&ptable.lock);
    for(p = ptable.proc; p < &ptable.proc[NPROC]; p++){
      if(p->state != RUNNABLE)
        continue;
    
    if(p->SchedulingPolicy == 2){

    
      highP = p;
      //選擇有最高priority的process
      for(p1 = ptable.proc; p1 < &ptable.proc[NPROC]; p1++){
        if(p1->state != RUNNABLE)
          continue;
        if( highP->priority > p1->priority ) //數字越小，priority越前面
          highP = p1;
      }
      p = highP;
     }
      // Switch to chosen process.  It is the process's job
      // to release ptable.lock and then reacquire it
      // before jumping back to us.
      c->proc = p;
      switchuvm(p);
      p->state = RUNNING;

      swtch(&(c->scheduler), p->context); //當process結束或是process的使用時間到時就會return
      switchkvm();

      // Process is done running for now.
      // It should have changed its p->state before coming back.
      c->proc = 0;
    }
    release(&ptable.lock);

}
}

最後，結合上次作業做的system call APS()跟CPP()做出PC.c Process Controller

















































#include "types.h"
#include "stat.h"
#include "user.h"
#include "fcntl.h"

int
main(int argc, char *argv[])
{ 
  char state;
  while(1){
    printf(1, "\nEnter '1' to change scheduling process:\n");
    printf(1, "Enter '2' to show all process state:\n");
    printf(1, "Enter '3' to change process priority:\n");
    printf(1, "Enter '4' to exit:\n");
    scanf(0, "%c", &state);
    
    if(state == '1'){
      char t;
      printf(1, "Please enter 'P' for priority scheduling, 'R' for round robin scheduling: ");
      scanf(0, "%c", &t);
      if(t == 'R'){
        CSP(1);
      }
      if(t == 'P'){
        CSP(2);
      }
    }
    
    if(state == '2'){
      APS();
    }
    
    if(state == '3'){
       int pid, newp;
       printf(1, "Please enter the pid of the process that you want to change priority: ");
       scanf(0, "%d", &pid);
       printf(1, "Please enter the new priority of the process: ");
       scanf(0, "%d", &newp);
       CPP( pid, newp );
    }
    
    if(state == '4'){
      exit();
    }
    
  }
  exit();
}
