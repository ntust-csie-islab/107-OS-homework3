#include "types.h"

int readline(int fd, char* buffer)
{
	char c;
	int i=0,n;
	while(1)
	{
		n=read(fd,&c,1);
		if(n<0)
			continue;
		else if(n==0||c=='\n')
			break;
		buffer[i++] = c;
	}
	return i;
}

int len(int n)
{
	int i=1;
	while(n>9)
		n = n/10,i++;
	return i;
}

void pitoa(int n, char* buf)
{
	int i = len(n);
	while(i-->=0)
	{
		buf[i] = n%10+'0';
		n /= 10;
	}
}

int combination(int n, int k)
{
	int a = (k>(n-k)) ? k:(n-k);
	int b = (k>(n-k)) ? (n-k):k;
	int ans = 1;
	for(int i=a+1;i<=n;i++)
		ans*=i;
	while(b)
		ans/=b--;
	return ans;
}

void comb(int** round,int* buf, int* p, int s, int num, int n, int k, int l)
{
	if(k==0)
	{
		for(int i=0;i<num;i++)
			*(*(round+*(p))+i) = buf[i];
		(*p)++;
		return;
	}
	for(int i=0;i<=n-k;i++)
	{
		buf[l++] = s+i;
		comb(round,buf,p,s+i+1,num,n-i-1,k-1,l);
		l--;
	}
}

struct player
{
	int id;
	int score;
	int rank;
};

void swap(int *x, int *y)
{
	int tmp = *x;
	*x = *y;
	*y = tmp;
}

void heapify(int arr[], int n, int i)
{
	int largest = i;
	int l = 2*i+1;
	int r = 2*i+2;
	if(l<n&&arr[l]>arr[largest])
		largest = l;
	if(r<n&&arr[r]>arr[largest])
		largest = r;
	if(largest != i)
	{
		swap(&arr[i],&arr[largest]);
		heapify(arr,n,largest);
	}
}

void heapSort(int arr[], int n)
{
	for(int i=n/2-1;i>=0;i--)
		heapify(arr,n,i);
	for(int i=n-1;i>=0;i--)
	{
		swap(&arr[0],&arr[i]);
		heapify(arr,i,0);
	}
}

void ranking(struct player* player,int nplayer)
{
	int *score = (int*)malloc(sizeof(int)*nplayer);
	for(int i=0;i<nplayer;i++)
		score[i] = player[i].score;
	heapSort(score,nplayer);
	for(int i=0;i<nplayer;i++)
	{
		for(int j=nplayer-1;j>=0;j--)
		{
			if(player[i].score==score[j])
			{
				player[i].rank = nplayer-j;
				break;
			}
		}
	}
	free(score);
}

uint random(void)
{
	//https://stackoverflow.com/questions/1167253/implementation-of-rand
	//static unsigned int z1 = 12345, z2 = 12345, z3 = 12345, z4 = 12345;
	int x = uptime()+getpid();
	unsigned int z1 = 12345+x, z2 = 12345+x, z3 = 12345+x, z4 = 12345+x;
	unsigned int b;
	b  = ((z1 << 6) ^ z1) >> 13;
	z1 = ((z1 & 4294967294U) << 18) ^ b;
	b  = ((z2 << 2) ^ z2) >> 27; 
	z2 = ((z2 & 4294967288U) << 2) ^ b;
	b  = ((z3 << 13) ^ z3) >> 21;
	z3 = ((z3 & 4294967280U) << 7) ^ b;
	b  = ((z4 << 3) ^ z4) >> 12;
	z4 = ((z4 & 4294967168U) << 13) ^ b;			  
	return (z1 ^ z2 ^ z3 ^ z4) / 2;
}

void heapSort(int arr[], int n)
{
	for(int i=n/2-1;i>=0;i--)
		heapify(arr,n,i);
	for(int i=n-1;i>=0;i--)
	{
		swap(&arr[0],&arr[i]);
		heapify(arr,i,0);
	}
}

void ranking(struct player* player)
{
	int win[4];
	for(int i=0;i<4;i++)
		win[i] = player[i].win;
	heapSort(win,4);
	for(int i=0;i<4;i++)
	{
		for(int j=3;j>=0;j--)
		{
			if(player[i].win==win[j])
			{
				player[i].rank = 4-j;
				break;
			}
		}
	}
}
