🙂 使用情境說明(包含流程圖)

我們在作業2實作了類似select的system call，因此有了I/O multiplexing的功能，在作業3，我們利用這個system call，寫了一個模擬的game system，包含game system、host和player三個程式。

game system主要的功能是開host、指派player和統計最後的排名，game system可以透過參數決定要開幾個host和總共有多少個player，然後fork並exec執行host，game system與host間透過pipe傳輸，game system可以指派player給host，host在一場遊戲結束後，回傳排名給game system。而host的主要功能是舉辦遊戲，host在接收到game system的指派後，會fork並exec執行player，host和player也是透過pipe溝通，我們是用最簡單的猜數字遊戲來模擬，因此host必須給player數字的範圍，player回傳猜的數字，host在10局結束後，就會統計排名給game system。而player的功能就是猜數字，這裡透過亂數的方式來模擬。

我們在作業2實作的selectpiperead這個system call在這個應用中，主要用在game system，game system必須與多個host溝通，指派玩家給host和從host接收排名資訊，blocking mode在讀不到時會blocking，而non-blocking mode會造成busy waiting，因此非常適合使用I/O multiplexing。

#include "types.h"
#include "stat.h"
#include "user.h"

int readline(int fd, char* buffer)
{
	char c;
	int i=0,n;
	while(1)
	{
		n=read(fd,&c,1);
		if(n<0)
			continue;
		else if(n==0||c=='\n')
			break;
		buffer[i++] = c;
	}
	return i;
}

int len(int n)
{
	int i=1;
	while(n>9)
		n = n/10,i++;
	return i;
}

void pitoa(int n, char* buf)
{
	int i = len(n);
	while(i-->=0)
	{
		buf[i] = n%10+'0';
		n /= 10;
	}
}

int combination(int n, int k)
{
	int a = (k>(n-k)) ? k:(n-k);
	int b = (k>(n-k)) ? (n-k):k;
	int ans = 1;
	for(int i=a+1;i<=n;i++)
		ans*=i;
	while(b)
		ans/=b--;
	return ans;
}

void comb(int** round,int* buf, int* p, int s, int num, int n, int k, int l)
{
	if(k==0)
	{
		for(int i=0;i<num;i++)
			*(*(round+*(p))+i) = buf[i];
		(*p)++;
		return;
	}
	for(int i=0;i<=n-k;i++)
	{
		buf[l++] = s+i;
		comb(round,buf,p,s+i+1,num,n-i-1,k-1,l);
		l--;
	}
}

struct player
{
	int id;
	int score;
	int rank;
};

void swap(int *x, int *y)
{
	int tmp = *x;
	*x = *y;
	*y = tmp;
}

void heapify(int arr[], int n, int i)
{
	int largest = i;
	int l = 2*i+1;
	int r = 2*i+2;
	if(l<n&&arr[l]>arr[largest])
		largest = l;
	if(r<n&&arr[r]>arr[largest])
		largest = r;
	if(largest != i)
	{
		swap(&arr[i],&arr[largest]);
		heapify(arr,n,largest);
	}
}

void heapSort(int arr[], int n)
{
	for(int i=n/2-1;i>=0;i--)
		heapify(arr,n,i);
	for(int i=n-1;i>=0;i--)
	{
		swap(&arr[0],&arr[i]);
		heapify(arr,i,0);
	}
}

void ranking(struct player* player,int nplayer)
{
	int *score = (int*)malloc(sizeof(int)*nplayer);
	for(int i=0;i<nplayer;i++)
		score[i] = player[i].score;
	heapSort(score,nplayer);
	for(int i=0;i<nplayer;i++)
	{
		for(int j=nplayer-1;j>=0;j--)
		{
			if(player[i].score==score[j])
			{
				player[i].rank = nplayer-j;
				break;
			}
		}
	}
	free(score);
}

int main(int argc, char* argv[])
{
	if(argc!=3)
		exit();
	int nhost = atoi(argv[1]);
	int nplayer = atoi(argv[2]);
	char buf[64];
	int pid,iround=0,ifround=0,n;

	int stin = dup(0);
	int stout = dup(1);

	struct player *player;
	player = malloc(sizeof(struct player)*nplayer);
	memset(player,0,sizeof(struct player)*nplayer);
	for(int i=0;i<nplayer;i++)
		player[i].id = i+1;

	int **round;
	int nround = combination(nplayer,4);
	round = (int**)malloc(sizeof(int*)*nround);
	for(int i=0;i<nround;i++)
		round[i] = (int*)malloc(sizeof(int)*4);
	int *ibuf = malloc(sizeof(int)*4);
	int z = 0;
	int *p = &z;
	comb(round,ibuf,p,1,4,nplayer,4,0);

	int **pfd;
	pfd = (int**)malloc(sizeof(int*)*nhost*2);
	for(int i=0;i<nhost*2;i++)
	{
		pfd[i] = (int*)malloc(sizeof(int)*2);
		pipe(pfd[i]);
	}

	for(int i=0;i<nhost;i++)
	{
		close(0);
		close(1);
		dup(pfd[i][0]);
		dup(pfd[i+nhost][1]);

		pid = fork();
		if(pid==0)
		{
			for(int i=3;i<pfd[nhost-1][1];i++)
				close(i);
			char id_[16];
			pitoa(i,id_);
			char* arg[3] = {"host",id_,0};
			exec(arg[0],arg);
		}
		else
		{
			close(pfd[i][0]);
			close(pfd[i+nhost][1]);
		}
		if(iround<nround)
		{
			printf(pfd[i][1],"%d %d %d %d\n",round[iround][0],round[iround][1],round[iround][2],round[iround][3]);
			iround++;
		}
	}

	int *fdsptr = (int*)malloc(sizeof(int)*nhost);

	while(ifround<nround)
	{
		for(int i=0;i<nhost;i++)
			fdsptr[i] = pfd[i+nhost][0];
		n = selectpiperead(fdsptr,nhost,100);

		if(n>0)
		{
			for(int i=0;i<nhost;i++)
			{
				if(fdsptr[i]>=0)
				{
					memset(buf,0,64);
					for(int j=0;j<4;j++)
					{
						readline(fdsptr[i],buf);
						int id_ = atoi(buf);
						int rk_ = atoi(buf+len(id_)+1);
						player[id_-1].score += 5-rk_;
					}
					ifround++;
					if(iround<nround)
					{
						printf(pfd[i][1],"%d %d %d %d\n",round[iround][0],round[iround][1],round[iround][2],round[iround][3]);
						iround++;
					}
				}
			}
		}
	}

	for(int i=0;i<nhost;i++)
	{
		printf(pfd[i][1],"%d %d %d %d\n",0,0,0,0);
		wait();
	}

	ranking(player,nplayer);

	printf(stout,"id\trank\n");	
	for(int i=0;i<nplayer;i++)
	{
		printf(stout,"%d\t%d\n",player[i].id,player[i].rank);
	}

	for(int i=0;i<nhost;i++)
	{
		close(pfd[i][1]);
		close(pfd[i+nhost][0]);
	}

	free(player);
	free(fdsptr);
	for(int i=0;i<nround;i++)
		free(round[i]);
	free(round);
	for(int i=0;i<nhost*2;i++)
		free(pfd[i]);
	free(pfd);
	return 0;
}

	{
		n=read(fd,&c,1);

		else if(n==0||c=='\n')

	}

}

int len(int n)
{
	int i=1;
	while(n>9)
		n = n/10,i++;
	return i;

uint random(void)
{
	//https://stackoverflow.com/questions/1167253/implementation-of-rand
	//static unsigned int z1 = 12345, z2 = 12345, z3 = 12345, z4 = 12345;
	int x = uptime()+getpid();
	unsigned int z1 = 12345+x, z2 = 12345+x, z3 = 12345+x, z4 = 12345+x;
	unsigned int b;
	b  = ((z1 << 6) ^ z1) >> 13;
	z1 = ((z1 & 4294967294U) << 18) ^ b;
	b  = ((z2 << 2) ^ z2) >> 27; 
	z2 = ((z2 & 4294967288U) << 2) ^ b;
	b  = ((z3 << 13) ^ z3) >> 21;
	z3 = ((z3 & 4294967280U) << 7) ^ b;
	b  = ((z4 << 3) ^ z4) >> 12;
	z4 = ((z4 & 4294967168U) << 13) ^ b;			  
	return (z1 ^ z2 ^ z3 ^ z4) / 2;
}

int rand(int start, int end)
{
	return start+1+(random()%(end-start-1));
}

void pitoa(int n, char* buf)
{
	int i = len(n);
	while(i-->=0)
	{
		buf[i] = n%10+'0';
		n /= 10;
	}
}

struct player
{
	int id;
	int win;
	int rank;
};

void swap(int *x, int *y)
{
	int tmp = *x;
	*x = *y;
	*y = tmp;
}

void heapify(int arr[], int n, int i)
{
	int largest = i;
	int l = 2*i+1;
	int r = 2*i+2;
	if(l<n&&arr[l]>arr[largest])
		largest = l;
	if(r<n&&arr[r]>arr[largest])
		largest = r;
	if(largest != i)
	{
		swap(&arr[i],&arr[largest]);
		heapify(arr,n,largest);
	}
}

void heapSort(int arr[], int n)
{
	for(int i=n/2-1;i>=0;i--)
		heapify(arr,n,i);
	for(int i=n-1;i>=0;i--)
	{
		swap(&arr[0],&arr[i]);
		heapify(arr,i,0);
	}
}

void ranking(struct player* player)
{
	int win[4];
	for(int i=0;i<4;i++)
		win[i] = player[i].win;
	heapSort(win,4);
	for(int i=0;i<4;i++)
	{
		for(int j=3;j>=0;j--)
		{
			if(player[i].win==win[j])
			{
				player[i].rank = 4-j;
				break;
			}
		}
	}
}

int main(int argc, char* argv[])
{
	if(argc!=2)
		exit();

	int id = atoi(argv[1]);
	char buf[64];
	int n,pid;
	struct player player[4];

	int stin = dup(0);
	int stout = dup(1);

	while(1)
	{
		memset(buf,0,64);
		n = readline(stin,buf);
		if(n==0)
			break;
		int c = 0;
		for(int i=0;i<4;i++)
		{
			player[i].id = atoi(buf+c);
			player[i].win = 0;
			player[i].rank = 0;
			c += len(player[i].id)+1;
		}
		if(player[0].id==0&&player[1].id==0&&player[2].id==0&&player[3].id==0)
			break;

		int pfd[4*2][2];
		for(int i=0;i<4*2;i++)
			pipe(pfd[i]);

		for(int i=0;i<4;i++)
		{
			close(0);
			close(1);
			dup(pfd[i][0]);
			dup(pfd[i+4][1]);
			pid = fork();

			if(pid==0)
			{
				for(int i=2;i<pfd[7][1];i++)
					close(i);
				char* arg[2] = {"player",0};
				exec(arg[0],arg);
			}
			else
			{
				close(pfd[i][0]);
				close(pfd[i+4][1]);
			}
		}

		for(int j=0;j<10;j++)
		{
			int start=0,end=1000;
			int ans = rand(0,1000);
			int receive,i;

			for(i=0;;i=(i+1)%4)
			{
				printf(pfd[i][1],"%d %d\n",start,end);
				memset(buf,0,64);
				readline(pfd[i+4][0],buf);
				receive = atoi(buf);
				if(receive==ans)
					break;
				else if(receive>ans)
					end = receive;		
				else if(receive<ans)
					start = receive;
			}
			player[i].win++;
		}

		ranking(player);

		for(int i=0;i<4;i++)
		{
			printf(stout,"%d %d\n",player[i].id,player[i].rank);
			printf(pfd[i][1],"%d %d\n",0,0);
			wait();
			close(pfd[i][1]);
			close(pfd[i+4][0]);
		}
	}
	return 0;
}

😎 結論

I/O multiplexing通常用於單個process處理多個I/O的情形，其需要監聽多個I/O的connection，像是socket、FIFO或pipe的相關應用，因xv6作業系統較精簡，要做比較實用的application的話，需要再增加kernal的service，因此這裡透過模擬一個game system來展示一個process監聽多個I/O的情況。在實作時，因為要處理process之間的溝通和一些I/O的redirect，因此必須特別注意，不然就會有許多bug。
