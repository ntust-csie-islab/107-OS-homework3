static void
getc(int fd, char ** buffer) {
	int  i = 0;
	char buf[256];	// read input into buffer
	while(read(fd, &buf[i], 1)){
		if((buf[i] == '\n' || buf[i] == ' ' || buf[i] == '\t' || buf[i] == '\r' || buf[i] == '\v' || buf[i] == '\f')){
			break;
		}
		else{
			i++;
		}
	}
	buf[i] = '\0';
	*(buffer) = buf;
}

uint atoi_octal(char *buf, int start) {   
	uint res = 0 ;

	for(; buf[i] != '\0' ; i++){

// %c : character
// %d : signed integer
// %s : string
// %o : Octal number
void scanf(int fd, char * fmt, ...) {
	int i = 0;
	char c;
	char * buf = "";
	int count_args = 1;
	uint var = *((uint*)(void*)&fmt + count_args);
	count_args++;
	for(i = 0; fmt[i]; i++){
		c = fmt[i] & 0xff;
		switch(c){
			case 'c':
				getc(fd, &buf);
				*(char *)(var) = buf[0];
				buf = "";
				var = *((uint*)(void*)&fmt + count_args);
				count_args++;
				break;
			case 'd':
				getc(fd,&buf);
				*(int *)(var) = atoi_decimal(buf);
				buf = "";
				var = *((uint*)(void*)&fmt + count_args);
				count_args++;
				break;
			case 's':
				getc(fd, &buf);
				strcpy((char *)var, buf);
				var = *((uint*)(void*)&fmt + count_args);
				count_args++;
				buf = "";	
				break;
			case 'o':
				getc(fd, &buf);
				*(uint *)(var) = atoi_octal(buf, 0);
				buf = "";
				var = *((uint*)(void*)&fmt + count_args);
				count_args++;
				break; 
		}
	}
}

新增這次作業的應用程式，先建立table後再根據使用者的輸入查詢是否可以胡牌

#include "types.h"
#include "user.h"

#define SIZE 0444444444

short* table;
int melds[17], e[9];

int have(int m, int s)
{
	for (int i = 0; i < 9; i++)
		if (((m >> i * 3) & 7) < ((s >> i * 3) & 7))
			return 0;
	return 1;
}
void set(int n, int d, int p)
{
	int i;
	if (n)
	{
		for (i = 0; i<17; i++)
			if (have(p, melds[i]))
				set(n - 1, d + melds[i], p - melds[i]);
	}
	else
	{
		table[d] |= 1;
		for (i = 0; i<9; i++)
			if (have(p, e[i]))
				table[d + e[i]] |= 3;
	}
}

int init()
{
	int i;
	table = (short*)calloc(SIZE, sizeof(short));
	if (!table)
		return 0;
	
	melds[0] = 0;;
	for (i = 0; i<9; i++)
		melds[i+1] = (3 << i * 3);
	for (i = 0; i<7; i++)
		melds[i+1+9] = (73 << i * 3);
	for (i = 0; i<9; i++)
		e[i] = (2 << i * 3);
	set(4, 0, SIZE);
	
    printf(1, "init done\n");
	return 1;
}
short clac(int a, int b, int c)
{
	if(table[a]&table[b]&table[c]&1)
		return ((table[a]|table[b]|table[c])&2);
	return -1;
}
int main(int argc, char *argv[])
{
    if(!init())
        exit();
    
    int a ,b, c;
	printf(1, "輸入萬 筒 條各數字張數\n");
	printf(1, "例子: 111002 300 11100 => 一萬一萬四萬五萬六萬 三筒三筒三筒 三條四條五條\n");
    scanf(0, "%o %o %o", &a, &b, &c);
    
    while(1)
    {
		if(clac(a, b, c) > 0)
			printf(1, "\n可以胡牌\n");
		else
			printf(1, "\n無法胡牌\n");

}

😎 結論

記憶體管理是OS一個很重要的問題，像是程式一開始執行時，會alloc許多記憶體，但並不會馬上使用，所以透過修改sbrk()的機制來增進效能，不過效果仍有限。可能能透過swap等方法來解決。
而雖然alloc()的這個程式的架構並不龐大，但卻非常的實用，不只顯示了基本資料型態的大小，也幫忙計算出了總記憶體大小，減少了許多計算的時間，非常的方便實用。
這次作業實作中，碰到了許多問題，許多常用的lib xv6都沒有，像是加入一個scanf就花費不少時間，不只要trace code，還要了解整體架構才能實現，更重要的是，google能力要好，參考前人的想法並融會貫通，而不是只靠課堂上微薄的知識，是這幾次作業的深刻體悟。
